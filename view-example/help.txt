# View Model
<script>
    function bindInput(vm, prop, inputElement) {
    // Update input when VM changes
    vm.subscribe(prop, (value) => {
        if (inputElement.value !== value) {
        inputElement.value = value;
        }
    });

    // Update VM when input changes
    inputElement.addEventListener("input", (e) => {
        vm.set(prop, e.target.value);
    });

    // Initialize
    inputElement.value = vm.get(prop) || "";
    }

    function bindDisabled(vm, prop, buttonElement) {
    // Update button disabled state when VM changes
    vm.subscribe(prop, (value) => {
        buttonElement.disabled = !!value;
    });

    // Initialize
    buttonElement.disabled = !!vm.get(prop);
    }
</script>
<script>
    const vm = createViewModel({
    query: "",
    isDisabled: true,
    });

    const input = document.getElementById("search-bar");
    const button = document.getElementById("search-button");

    // Bind properties
    bindInput(vm, "query", input);
    bindDisabled(vm, "isDisabled", button);

    // Example derived logic
    vm.subscribe("query", (value) => {
    vm.set("isDisabled", value.trim().length === 0);
    });
</script>
